sites-enabled is:

%  more /etc/nginx/sites-available/smarterplaylists

server {
    location /api {
        proxy_pass http://localhost:8000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
    }
    location / {
        alias  /home/www/smarterplaylists/;
    }
}
